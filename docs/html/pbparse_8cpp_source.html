21  void Parse::start(String filename)
22  {
23  File source = SD.open(filename.c_str());
24  if(!source)
25  {
26  #ifdef CRASH_MSG_DETAIL
27  Crash::forceHalt("PB runner could not open file.");
28  #else
29  Crash::forceHalt("Ex006");
30  #endif
31  }
32  Stack::push(filename);
33  const char terminator = ';';
34  while(source.available())
35  {
36  String cmd = "";
37  while(char(source.peek()) != terminator)
38  {
39  cmd += char(source.read());
40  }
41  source.read();
42  Parse::run(cmd);
43  }
44  source.close();
45  Stack::pop(filename);
46  }
50  void Parse::run(String line)
51  {
52  line.trim();
53  if(line.startsWith("//"))
54  {
55  // Do nothing, this is a comment
56  }
57  else
58  {
59  String opcode = Tokenizer::TKgetToken(line, 0);
60  if(opcode == "PRINT") Parse::Opcode::IO::PRINT(Tokenizer::TKgetToken(line, 1));
61  else if(opcode == "PRINTV") Parse::Opcode::IO::PRINTV(Tokenizer::TKgetToken(line, 1));
62  else if(opcode == "ADDS") Parse::Opcode::Var::Strings::ADDS(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2), Tokenizer::TKgetToken(line, 3));
63  else if (opcode == "GETC") Parse::Opcode::Var::Strings::GETC(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2), Tokenizer::TKgetToken(line, 3));
64  else if(opcode == "NEWPRINT") Parse::Opcode::IO::NEWPRINT();
65  else if(opcode == "WAIT") Parse::Opcode::System::WAIT(Tokenizer::TKgetToken(line, 1));
66  else if(opcode == "IFE") Parse::Opcode::Logic::IFE(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2), Tokenizer::TKgetToken(line, 3));
67  else if(opcode == "IFNE") Parse::Opcode::Logic::IFNE(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2), Tokenizer::TKgetToken(line, 3));
68  else if(opcode == "CREATESWAP") Parse::Opcode::System::SWAP::CREATESWAP();
69  else if(opcode == "IO") Parse::Opcode::IO::Ports::IO(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2));
70  else if(opcode == "NEW") Parse::Opcode::System::SWAP::NEW(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2));
71  else if(opcode == "DELETE") Parse::Opcode::System::SWAP::DELETE(Tokenizer::TKgetToken(line, 1));
72  else if(opcode == "SET") Parse::Opcode::System::SWAP::SET(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2));
73  else if(opcode == "EXTLOAD") Parse::Opcode::System::EXTLOAD(Tokenizer::TKgetToken(line, 1));
74  else if(opcode == "FILEWRITE") Parse::Opcode::IO::Files::FILEWRITE(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2));
75  else if(opcode == "FILEREAD") Parse::Opcode::IO::Files::FILEREAD(Tokenizer::TKgetToken(line, 1), Tokenizer::TKgetToken(line, 2));
76  else if(opcode == "END") Parse::Opcode::System::END();
77  else
78  {
79  #ifdef CRASH_MSG_DETAIL
80  Crash::forceHalt("Error, unrecognized command.");
81  #else
82  Crash::forceHalt("Ex007");
83  #endif
84  }
85  }
